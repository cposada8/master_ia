def CI(S,COSTES):
  VALOR = 0
  #Valores establecidos
  for i in range(len(S)):
    VALOR += COSTES[i][S[i]]

  #Estimacion
  for i in range( len(S), len(COSTES)   ):
    VALOR += min( [ COSTES[j][i] for j in range(len(S), len(COSTES))  ])
  return VALOR

def CS(S,COSTES):
  VALOR = 0
  #Valores establecidos
  for i in range(len(S)):
    VALOR += COSTES[i][S[i]]

  #Estimacion
  for i in range( len(S), len(COSTES)   ):
    VALOR += max( [ COSTES[j][i] for j in range(len(S), len(COSTES))  ])
  return VALOR

def costo_solucion(S,COSTES):
  VALOR = 0
  #Valores establecidos

  return VALOR

def asignacion_tareas_fuerza_bruta(costes):
  n = len(costes)
  min_cost = np.inf
  sol_opt = None
  # obtengo todas las permutaciones posibles
  permutations = itertools.permutations(range(n))
  for option in permutations:
    costo_actual = costo_solucion(option, costes)
    if costo_actual < min_cost:
      min_cost = costo_actual
      sol_opt = option
  return sol_opt

def step_gradiente_descendiente(df, w, alpha = 0.05):
    w = w - alpha*df(w)
    return w

def gradiente_descendiente(f, df, w, max_iter = 100, alpha = 0.05):
  # esta funcin ejecuta varios pasos del algoritmo de gradiente descendiente
  # y retornar:
  # w: el vector de solucin luego de ejecutar los steps de gradiente descendiente
  # ws: un vector con todas las w que encontr en cada step (incluida la solucin original)
  ws = [w]
  for i in range(max_iter):
    w = step_gradiente_descendiente(df, w, alpha)
    ws.append(w)
  return w, ws

def f2(X):
  # funcin con muchos mnimos
  return np.sin(1/2 * X[0]**2 - 1/4 * X[1]**2 + 3) * np.cos(2 * X[0] + 1 - np.exp(X[1]))

def numeric_derivative(f, X, h=0.001):
  # funcin para aproximar la derivada de una funcin dada evaluada en un punto
  X = np.array(X)
  grad = np.zeros(len(X))
  for i in range(len(X)):
    grad[i] = (f(X+h) - f(X)) / h
  return grad

def d_f2(X):
  return numeric_derivative(f2, X)
